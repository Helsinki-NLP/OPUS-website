خۆشیمان دەداتێ ؟ لەگەڵ داوای لێبوردنم بۆ ماثیو ڕیچاردی هاوڕێم مۆڵێکی پڕ لە زێن مۆنکس قازانجێکی تایبەتی نابێت لەبەرئەوەی خەڵک ئەوەندەیان شتومەک ناوێت دەمەوێت پێشنیاری ئەوە بکەم کە دڵخۆشی دروستکراو هەر بەردەوام و ڕاستی دەبێت بەهەمان شێوەی ئەو دڵخۆشییەی ئێوە بەدوایدا دەگەڕێن کاتێ بەتەواوی ئەوەتان دەست دەکەوێت کە دەتانەوێت ئێستا من زانایەکم ، ئەم زانیاریانەتان پێدەڵێم نەک بەشێوەیەکی ئەدەبیاتیانە بەڵکو بەهەندێک داتا با لەسەرەتادا ئەم نمونە ئەزمونکراوەتان نیشان بدەم بۆ ئەوەی دڵخۆشی دروستکراوتان بۆ وەسف بکات ئەم داتایانە هی کۆمەڵێک کەسی بەساڵاچون ، هی من نین ئەمە نمونەیەکی ٥٠ ساڵەیە کە ناونراوە نمونەی هەڵبژاردنی سەربەخۆ زۆر ئاسانە شەش شت دەهێنیت لەکەسەکە دەپرسیت ڕێکیان بخە بە گوێرەی ئەوەی زۆرتر بەدڵتە تا کەمتر لەم حاڵەتەدا ئەوەی کە دەمەوێت پێتان بڵێم لەسەر ئەم ئەزمونە ، شتێکە هەموان بەکاری دەهێنن پرنتەری مۆنێتە کەواتە هەمووتان دەتوانن ئەم پرنتەرانە ڕیز بکەن لەوەوە دەست پێبکەن کە زۆر بەدڵیانە تا ئەوەی کە کەمترین ڕێژە بەدڵیانە ئێستا ئێمە هەڵبژاردنتان دەدەینێ هەندێک پرنتەری ترمان مابو لە کۆگاکەدا یەکێکی ترتان دەدەینێ وەکو دیاری بیبەنەوە بۆ ماڵەوە ئەوەی کە ماومانە تەنها جۆرەکانی پرنتەری ژمارە سێ و چوارە بە کەسەکە دەڵێین ئەمە هەڵبژاردنێکی تۆزێک قورسە لەبەرئەوەی هیچ کام لەم پرنتەرانە لەوانە نین کە زۆر بەدڵت بوە بەڵام بەشێوەیەکی سروشتی زۆرینەی خەڵک ژمارە سێ هەڵدەبژێرن لەبەرئەوەی یەک تۆز زیاتر بەدڵیان بووە لە پرنتەری ژمارە چوار دوای ماوەیەک .. ئیتر ئایا ماوەکە ١٥ خولەک یان ١٥ ڕۆژ بێت هەمان بەرهەم لەبەردەمی کەسەکاندا دانرا داوا لە کەسەکان کرا بۆ جارێکی تر بەرهەمەکان ڕێکبخەنەوە ئێستا پێمان بڵێ چەند بەدڵتن چی ڕوویدا ؟ سەیرکەن چۆن دڵخۆشی دروستکراوە ئەم ئەنجامەمان دەستکەوت لەهەموو جارێک لەم تاقیکردنەوەیە ئێوە هەمووتان دەبینن چۆن دڵخۆشی دروستکراوە دەتانەوێت جارێکی تر بیبینن ؟ دڵخۆشی ئەوەی دەستم کەوتوە زۆر باشترە لەوەی چاوەڕێم دەکرد ئەوەی کە حەزم لێیبو و دەستم نەکەوت زۆر بەرهەمێکی خراپە ( پێکەنین ) ، ئەوەیە دڵخۆشی دروستکراو ئەی کەواتە وەڵامی ئاسایی ئێمە چی دەبێت بۆ ئەو کەسانە ؟ بەڵێ وایە ئێستا ئەمە ئەو تاقیکردنەوەیەیە کە ئێمە کردمان وە هیوادارم ئەمە قەناعەت بە ئێوە بکات کە بەڵێ وایە ، وەڵامی ڕاست نیە ئەم تاقیکردنەوەیەمان لەگەڵ گروپێک نەخۆشدا ئەنجامدا کە تووشبون بە نەخۆشی بیرچونەوەیە کە زۆربەیان حاڵەتی کۆرساکۆف سیندرۆمە نەخۆشیەکە ئەوەیە کە زۆر دەخۆنەوە ناشتوانن بیرەوەری تازە دروستبکەن باشە ؟ ئەوان بیری منداڵییان دەمێنێت بەڵام ئەگەر بچیتە لایان و خۆتیان پێ بناسێنیت دواتر ژوورەکە بەجێبهێڵیت دواتر کە گەڕایتەوە بۆ لایان ئەوا ناتناسنەوە پرنتەرەکانمان بردە ئەو نەخۆشخانەیەی کە نەخۆشەکانمانی لێبوو داوامان لە نەخۆشەکان کرد ڕیزیان بکەن لەوەوە کە زۆرترین بەدڵیانە تا ئەوەی کەمترین دواتر پێمانوتن هەڵبژێرن لە نێوان سێ و چواردا وەکو هەموو کەسێکی تر ووتیان ئۆو سوپاس دکتۆر ئەمە شتێکی باشە من دەتوانم پرنتەرێکی تازە بەکاربهێنم پرنتەری ژمارە سێ دەبەم بۆمان ڕونکردنەوە کە ئەوەی ژمارە سێیان بۆ دەنێرین دواتر شتومەکەکانمان کۆکردەوەو لە ژوورەکە چوینە دەرەوە نیو کاتژمێرمان بژارد گەڕاینەوە ژوورەکە ووتمان سڵاو هاتینەوە نەخۆشەکان ووتیان ئاه دکتۆر ببورە من کێشەی بیرچونەوەم هەیە بۆیە لێرەم ئەگەر پێشتر تۆم بینیبێت بیرم نەماوە بەڕاست ، جیم تۆ منت بیر نیە ؟ من تۆزێک پێش ئێستا لێرەبووم لەگەڵ پرنتەرەکاندا ببورە دکتۆر هیچم بیر نیە کێشە نیە جیم تەنها دەمەوێت ئەمانەم بۆ ڕیز بکەیت لەو دانەیەوەی کە زۆرتر بەدڵتە بۆ ئەوەی کەمتر بەدڵتە دەبێت چیبکەین ؟ باشە ، با لەسەرەتادا سەیرێک بکەین و دڵنیا بین بەڵام ئەمانە تووشی بیرچونەوە بوون و داوامان لەم نەخۆشانە کردوە پێمان بڵێن کامەیان هی خۆیانە کامەیان هەڵبژاردوە جاری ڕابردوو ، کامە هی خۆیانە وە ئەنجامەکە چی بوو ؟ چیان کرد نەخۆشەکان وە ئەنجامەکە چی بوو ؟ چیان کرد نەخۆشەکانئەگەر ئەو تاقیکردنەوەیەم لەگەڵ ئێوەش بکردایە ئێوە دەتانزانی کامەتان هەڵبژاردبوو بەڵام لەگەڵ ئەم نەخۆشانەدا ئەوان هیچیان بیر نیە ناتوانن هەمان پرنتەر هەڵبژێرنەوە کە جاری پێشوو هەڵیان بژاردوە ئەمانە بەشێوەیەکی ئاسایی کۆنتڕۆڵ دەکرێن ، ئەوان دڵخۆشی دروستکراون ئەمە گۆڕانکاریەکەیە کە تۆمارکرا گۆڕانکاری لە نێوان یەکەمجاری ڕیزکردنەکەیان بۆ دووەمجار ڕێکخستنەکان دەریدەخەن ئەوەی کە ئێستا نیشانمدان سیحریانەیە ئێستا بەشێوەیەکی هێڵکاریانە نیشانتانی دەدەم ئەوەی کە ئێستا هەمەباشترە لەوەی کە بیرم لێدەکردەوە ، ئەوەی کە نیمە ئەو باشە نیە کە بیرم لێدەکردەوە ئەوانەش کە تووشی نەخۆشی بیرچونەوە بوون هەمان شتیان کردوە ئەم کەسانە حەزیان لەباشترەکەیە بەڵام نازانن ئەوەی کە هەیانە باشترەکەیە بەڵێ وایە ، وەڵامە ڕاستەکە نیە ئەم کەسانە چیان کرد کاتێک دڵخۆشیەکی دروستکراویان بەدەست هێنا ئەوان بەڕاستی گۆڕان کاریگەری و بیرکردنەوەیان گۆڕا تەنها واناڵین لەبەرئەوەی کە هەیانە بەڵکو لەبەرئەوەی نازان هەیانە ئێستا کاتێک دەروونناسەکان ئەو تووڵەت نیشان دەدەن ئێوە دەزانن کە ڕێژەیەکی مامناوەندیتان لەخەڵک نیشان دەدەن لەگەڵ ئەوەشدا ئێمە هەموومان ئەم سیستەمە دەرونیە بەرگریکارەمان هەیە ئەم توانایە بۆ دروستکردنی دڵخۆشی بەڵام هەندێکمان ئەم فێڵە بەشتر دەکەن وەک لەوانی تر هەندێک حاڵەتیش ڕێگە دەدەن هەر کەسێک بە کاریگەرانەتر بیانکات لە حاڵەتەکانی تر دەرکەوت کە سەربەخۆیی توانای بیرکردنەوەو هەڵبژاردن و گۆڕینی بۆچونەکەت هاوڕێی دڵخۆشی سروشتیە ، چونکە ڕێگەت دەدات هەڵبژێریت لە نێوان ئەو هەموو داهاتووە خۆشەداو لەگەڵ ئەوەی کە زۆرتر دڵت پێی خۆشە بەڵام ئازادی لەهەڵبژاردندا بۆ ئەوەی بۆچونەکەت بگۆڕیت ، دوژمنەکەی دڵخۆشی دروستکراوە منیش نیشانتان دەدەم بۆچی وایە دیلبێرت ، بێگومان پێشوەخت دەزانێت ئێوە وێنە کاریکاتێرەکان دەخوێننەوە لەم کاتەدا کە من قسە دەکەم یارمەتیدەرە تەکنەلۆجیەکەی دۆگبێرت ، چۆن دەتوانم دەستدرێژی بکەمە سەر پرنتەرەکەم دوای پرنت کردنی هەموو دۆکیۆمێنتێک پەڕەیەکی بەتاڵ پرنت دەکات باشە بۆچی ناڕازیت دەربارەی ئەوەی پەڕەی خۆڕایت دەست دەکەوێت ؟ خۆڕایی ؟ ئەی باشە تۆ هەمان ئەو پەڕانەم نادەیتێ کە هی خۆمن ؟ ئاه ئاخر سەیری کوالێتی پەڕە خۆڕاییەکان بکە بە بەراورد لەگەڵ پەڕە ئاساییەکانی خۆتدا تەنها درۆزنێک یان بێئەقڵێک دەڵێت ئەو دوانە وەک یەک وان ئۆو پاش ئەوەی باست کرد ، خۆ تۆزێک باشتر دیارە چی دەکەن ؟ من یارمەتی خەڵک دەدەم ئەو شتانە قبوڵ بکەن کە ناتوانن بیانگۆڕن لە ڕاستیدا سیستەمی بەرگریکردنی دەروونی بە باشترین شێوە کاردەکات لەکاتێکدا ئێمە بەتەواوەتی گیرمان خواردوە ئەمەیە جیاوازی نێوان ژوان و هاوسەرگیری ، وانیە ؟ مەبەستم ئەوەیە لەگەڵ کوڕیک دەچیتە ژوانێک پەنجە دەکات بە لووتیدا ، تۆش جارێکی تر لەگەڵیدا دەچیتەوە دەرەوە تۆ شووت بە پیاوێک کردوە پەنجە دەکات بە لووتیدا ؟ بەڵی بەڵام دڵێکی لە ئاڵتون هەیە باشە بەڵام دەست مەدە لە کێکە میوەییەکە باشە ؟ پێکەنین تۆ ڕێگەیەکت دۆزییەوە بۆ ئەوەی دڵخۆش بیت لەگەڵ ئەوەی ڕوویدا ئێستا ئەوەی دەمەوێت نیشانت بدەم ئەمەیە کە خەڵک ئەمە دەربارەی خۆیان نازانن نازانن ئەمە زیانی گەورەمان لێدەدات ئەمە تاقیکردنەوەیەکە لە زانکۆی هارڤارد کردومانە خولێکی فۆتۆگرافیمان دروستکرد ، فۆتۆگرافی ڕەش وسپی ڕێگەمان دا خوێندکارەکان بێن و فێر بن چۆن ژووری تاریک بەکاربهێنن لەبەرئەوە کامێرامان دانێ و چوونە خێوەتگاکان دوانزدە وێنەیان گرت ، هی باشترین مامۆستایان و ژوورەکانیان و شتەکانیان وە هەموو ئەو شتانەی تر کە دەیانویست لە هارڤاردا بیریان نەچێتەوە کامێراکانیان بۆ هێناینەوە ، ئێمەش پەڕەیەکی پەیوەندیمان دانا سەیریان کرد کامە دوو باشترین وێنەیە کە گرتوویانە ئێمەش بۆ ماوەی شەش کاتژمێر فێرمان کردوون چۆن ژووری تاریک بەکاربهێنن دوانیشیان تەقاندەوە دوو گۆڤاری هەشت بە دە ی زۆر جوانیان دروستکرد کە پڕبوون لەو شتانەی ماناداربوون بۆ خۆیان ، ئێمەش وتمان دەتانەوێت واز لە کامیان بهێنن ؟ وتیان دەبێت واز لە یەکێکیان بهێنین ؟ بەڵێ یەکێکیانمان دەوێت وەکو بەڵگەیەک بۆ پڕۆژەی گروپەکە جا لەبەر ئەوە دەبێت یەکێکیانمان بدەنێ ، دەبێت هەڵبژێریت تۆ یەکێک دەبەیت و منیش یەکێک تۆ یەکێک دەبەیت و منیش یەکێک لە حاڵەتێکیاندا بە خوێندکارەکان ووترا بەڵام تۆ دەزانیت ئەگەر بتەوێت ڕات بگۆڕیت ، من هەردەبێت یەکێکیانم لابێت لە چوار ڕۆژی داهاتوودا پێش ئەوەی بینێرم بۆ مەڵبەندەکان دڵخۆش دەبم ، ( پێکەنین ) ، بەڵێ مەڵبەندەکان دڵخۆش دەبم بەوەی یەکێکیانت لەگەڵدا بگۆڕمەوە لە ڕاستیدا دێم بۆ ژوورەکەت و یان تەنها ئیمەیڵەکەتم بدەرێ باشترە هەرکاتێک وویستت ڕات بگۆڕیت بەتەواوی ڕێگەپێدراویت بەڵام نیوەکەی تری خوێندکارەکان بەتەواوی پێچەوانەی ئەمەمان پێوتن هەڵبژێرە ، بەوبۆنەیەوە وێنەکان بە ئیمەیڵ دەنێرین بۆ ئینگلتەرا وێنەکەت دەفڕێت بەسەر زەریای ئەتڵەسیدا هەرگیز نایان بینیتەوە ئێستا لە نیوەی خوێندکارانی هەردوو حاڵەتەکە داواکرا پێشبینی بکەن تا چەند ئەو وێنانەی کە لایان دەمێنێتەوە بەدڵیانە لەگەڵ ئەوانەی کە جێی دەهێڵن خوێندکارەکانی تر گەڕێنرانەوە بۆ ژوورەکانیان تا سێ بۆ شەش ڕۆژی داهاتوویان بژارد لەسەر ئەوەی چەند وێنەیەکیان بەدڵەو لێیان ڕازین سەیرکەن چیمان دەستکەوت لەپێش هەموو شتێکدا خوێندکارەکان پێشبینی دەکەن ، چی ڕووبدات وا دەزانن لەوانەیە ئەو وێنانەی کە هەڵیان بژاردوە بەدڵیان دەبێت تۆزێک زیاتر لەوانەی کە بەجێیان هێشتوە بەڵام جیاوازی زۆریان نیە جیاوازیەکە کەمە و بەهەند وەرناگیرێت ئیتر ئایا لە حاڵەتێکی دوو ڕاییدا بووبن یان نا هەڵەن ، لەبەرئەوەی لەڕاستیدا ئەمە ڕوویدا
