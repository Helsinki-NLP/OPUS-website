    <P id="1">
        <s id="1.1">അദ്ധ്യായം 10</s>
    </P>
    <P id="2">
        <s id="2.1">1 “ആമേൻ, ആമേൻ, ഞാൻ നിങ്ങളോടു പറയുന്നു, ആട്ടിൻ തൊഴിത്തിൽ വാതിലൂടെ കടക്കാതെ വേറെ വഴിയായി കയറുന്നവൻ കള്ളനും കവർച്ചക്കാരനും ആകുന്നു.</s>
    </P>
    <P id="3">
        <s id="3.1">2 വാതിലൂടെ കടക്കുന്നവനോ ആടുകളുടെ ഇടയൻ ആകുന്നു.</s>
    </P>
    <P id="4">
        <s id="4.1">3 അവന്നു വാതിൽ കാവൽക്കാരൻ തുറന്നുകൊടുക്കുന്നു; ആടുകൾ അവന്റെ ശബ്ദം കേൾക്കുന്നു; തന്റെ ആടുകളെ അവൻ പേർ ചൊല്ലി വിളിച്ചു പുറത്തു കൊണ്ടുപോകുന്നു.</s>
    </P>
    <P id="5">
        <s id="5.1">4 തനിക്കുള്ളവയെ ഒക്കെയും പുറത്തുകൊണ്ടു പോയശേഷം അവൻ അവെക്കു മുമ്പായി നടക്കുന്നു; ആടുകൾ അവന്റെ ശബ്ദം അറിഞ്ഞു അവനെ അനുഗമിക്കുന്നു.</s>
    </P>
    <P id="6">
        <s id="6.1">5 അന്യന്മാരുടെ ശബ്ദം അറിയായ്കകൊണ്ടു അവ അന്യനെ അനുഗമിക്കാതെ വിട്ടു ഓടിപ്പോകും.</s>
    </P>
    <P id="7">
        <s id="7.1">6 ഈ സാദൃശ്യം യേശു അവരോടു പറഞ്ഞു; എന്നാൽ തങ്ങളോടു പറഞ്ഞതു ഇന്നതു എന്നു അവർ ഗ്രഹിച്ചില്ല.</s>
    </P>
    <P id="8">
        <s id="8.1">7 യേശു പിന്നെയും അവരോടു പറഞ്ഞതു: “ആമേൻ, ആമേൻ, ഞാൻ നിങ്ങളോടു പറയുന്നു: ആടുകളുടെ വാതിൽ ഞാൻ ആകുന്നു.</s>
    </P>
    <P id="9">
        <s id="9.1">8 എനിക്കു മുമ്പെ വന്നവർ ഒക്കെയും കള്ളന്മാരും കവർച്ചക്കാരും അത്രേ; ആടുകളോ അവരുടെ വാക്കു കേട്ടില്ല.</s>
    </P>
    <P id="10">
        <s id="10.1">9 ഞാൻ വാതിൽ ആകുന്നു; എന്നിലൂടെ കടക്കുന്നവൻ രക്ഷപ്പെടും; അവൻ അകത്തു വരികയും പുറത്തുപോകയും മേച്ചൽ കണ്ടെത്തുകയും ചെയ്യും.</s>
    </P>
    <P id="11">
        <s id="11.1">10 മോഷ്ടിപ്പാനും അറുപ്പാനും മുടിപ്പാനും അല്ലാതെ കള്ളൻ വരുന്നില്ല; അവർക്കു ജീവൻ ഉണ്ടാകുവാനും സമൃദ്ധിയായിട്ടു ഉണ്ടാകുവാനും അത്രേ ഞാൻ വന്നിരിക്കുന്നതു.</s>
    </P>
    <P id="12">
        <s id="12.1">11 ഞാൻ നല്ല ഇടയൻ ആകുന്നു; നല്ല ഇടയൻ ആടുകൾക്കു വേണ്ടി തന്റെ ജീവനെ കൊടുക്കുന്നു.</s>
    </P>
    <P id="13">
        <s id="13.1">12 ഇടയനും ആടുകളുടെ ഉടമസ്ഥനുമല്ലാത്ത കൂലിക്കാരൻ ചെന്നായ് വരുന്നതു കണ്ടു ആടുകളെ വിട്ടു ഓടിക്കളയുന്നു; ചെന്നായ് അവയെ പിടിക്കയും ചിന്നിച്ചുകളകയും ചെയ്യുന്നു.</s>
    </P>
    <P id="14">
        <s id="14.1">13 അവൻ കൂലിക്കാരനും ആടുകളെക്കുറിച്ചു വിചാരമില്ലാത്തവനുമല്ലോ.</s>
    </P>
    <P id="15">
        <s id="15.1">14 ഞാൻ നല്ല ഇടയൻ; പിതാവു എന്നെ അറികയും ഞാൻ പിതാവിനെ അറികയും ചെയ്യുന്നതുപോലെ ഞാൻ എനിക്കുള്ളവയെ അറികയും എനിക്കുള്ളവ എന്നെ അറികയും ചെയ്യുന്നു.</s>
    </P>
    <P id="16">
        <s id="16.1">15 ആടുകൾക്കു വേണ്ടി ഞാൻ എന്റെ ജീവനെ കൊടുക്കുന്നു.</s>
    </P>
    <P id="17">
        <s id="17.1">16 ഈ തൊഴുത്തിൽ ഉൾപ്പെടാത്ത വേറെ ആടുകൾ എനിക്കു ഉണ്ടു; അവയെയും ഞാൻ നടത്തേണ്ടതാകുന്നു; അവ എന്റെ ശബ്ദം കേൾക്കും; ഒരാട്ടിൻ കൂട്ടവും ഒരിടയനും ആകും.</s>
    </P>
    <P id="18">
        <s id="18.1">17 എന്റെ ജീവനെ വീണ്ടും പ്രാപിക്കേണ്ടതിന്നു ഞാൻ അതിനെ കൊടുക്കുന്നതുകൊണ്ടു പിതാവു എന്നെ സ്നേഹിക്കുന്നു.</s>
    </P>
    <P id="19">
        <s id="19.1">18 ആരും അതിനെ എന്നോടു എടുത്തുകളയുന്നില്ല; ഞാൻ തന്നേ അതിനെ കൊടുക്കുന്നു; അതിനെ കൊടുപ്പാൻ എനിക്കു അധികാരം ഉണ്ടു; വീണ്ടും പ്രാപിപ്പാനും അധികാരം ഉണ്ടു; ഈ കല്പന എന്റെ പിതാവിങ്കൽ നിന്നു എനിക്കു ലഭിച്ചിരിക്കുന്നു.</s>
    </P>
    <P id="20">
        <s id="20.1">19 ഈ വചനം നിമിത്തം യെഹൂദന്മാരുടെ ഇടയിൽ പിന്നെയും ഭിന്നത ഉണ്ടായി.</s>
    </P>
    <P id="21">
        <s id="21.1">20 അവരിൽ പലരും; അവന്നു ഭൂതം ഉണ്ടു; അവൻ ഭ്രാന്തൻ ആകുന്നു; അവന്റെ വാക്കു കേൾക്കുന്നതു എന്തിന്നു എന്നു പറഞ്ഞു.</s>
    </P>
    <P id="22">
        <s id="22.1">21 മറ്റു ചിലർ: ഇതു ഭൂതഗ്രസ്തന്റെ വാക്കല്ല; ഭൂതത്തിന്നു കുരുടന്മാരുടെ കണ്ണു തുറപ്പാൻ കഴിയുമോ എന്നു പറഞ്ഞു.</s>
    </P>
    <P id="23">
        <s id="23.1">22 അനന്തരം യെരൂശലേമിൽ പ്രതിഷ്ഠോത്സവം ആചരിച്ചു; അന്നു ശീതകാലമായിരുന്നു.</s>
    </P>
    <P id="24">
        <s id="24.1">23 യേശു ദൈവലായത്തിൽ ശലോമോന്റെ മണ്ഡപത്തിൽ നടന്നുകൊണ്ടിരുന്നു.</s>
    </P>
    <P id="25">
        <s id="25.1">24 യെഹൂദന്മാർ അവനെ വളഞ്ഞു: നീ എത്രത്തോളം ഞങ്ങളെ ആശിപ്പിക്കുന്നു? നീ ക്രിസ്തു എങ്കിൽ സ്പഷ്ടമായി പറക എന്നു അവനോടു പറഞ്ഞു.</s>
    </P>
    <P id="26">
        <s id="26.1">25 യേശു അവരോടു ഉത്തരം പറഞ്ഞതു: “ഞാൻ നിങ്ങളോടു പറഞ്ഞിട്ടുണ്ടു; എങ്കിലും നിങ്ങൾ വിശ്വസിക്കുന്നില്ല; എന്റെ പിതാവിന്റെ നാമത്തിൽ ഞാൻ ചെയ്യുന്ന പ്രവൃത്തികൾ എനിക്കു സാക്ഷ്യം ആകുന്നു.</s>
    </P>
    <P id="27">
        <s id="27.1">26 നിങ്ങളോ എന്റെ ആടുകളുടെ കൂട്ടത്തിലുള്ളവരല്ലായ്കയാൽ വിശ്വസിക്കുന്നില്ല. എന്റെ ആടുകൾ എന്റെ ശബ്ദം കേൾക്കുന്നു;</s>
    </P>
    <P id="28">
        <s id="28.1">27 ഞാൻ അവയെ അറികയും അവ എന്നെ അനുഗമിക്കയും ചെയ്യുന്നു.</s>
    </P>
    <P id="29">
        <s id="29.1">28 ഞാൻ അവെക്കു നിത്യജീവൻ കൊടുക്കുന്നു; അവ ഒരുനാളും നശിച്ചു പോകയില്ല; ആരും അവയെ എന്റെ കയ്യിൽ നിന്നു പിടിച്ചുപറിക്കയും ഇല്ല.</s>
    </P>
    <P id="30">
        <s id="30.1">29 അവയെ തന്നിരിക്കുന്ന എന്റെ പിതാവു എല്ലാവരിലും വലിയവൻ; പിതാവിന്റെ കയ്യിൽ നിന്നു പിടിച്ചുപറിപ്പാൻ ആർക്കും കഴികയില്ല</s>
    </P>
    <P id="31">
        <s id="31.1">30 ഞാനും പിതാവും ഒന്നാകുന്നു.”</s>
    </P>
    <P id="32">
        <s id="32.1">31 യെഹൂദന്മാർ അവനെ എറിവാൻ പിന്നെയും കല്ലു എടുത്തു.</s>
    </P>
    <P id="33">
        <s id="33.1">32 യേശു അവരോടു: “പിതാവിന്റെ കല്പനയാൽ ഞാൻ പല നല്ല പ്രവൃത്തികൾ നിങ്ങളെ കാണിച്ചിരിക്കുന്നു; അവയിൽ ഏതു പ്രവൃത്തിനിമിത്തം നിങ്ങൾ എന്നെ കല്ലെറിയുന്നു?” എന്നു ചോദിച്ചു.</s>
