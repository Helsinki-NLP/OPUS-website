וְשָׁפַט בֵּין הַגּוֹיִם וְהוֹכִיחַ לְעַמִּים רַבִּים וְכִתְּתוּ חַרְבוֹתָם לְאִתִּים וַחֲנִיתוֹתֵיהֶם לְמַזְמֵרוֹת לֹא־יִשָּׂא גוֹי אֶל־גּוֹי חֶרֶב וְלֹא־יִלְמְדוּ עוֹד מִלְחָמָה׃
וְיָדִין בֵּין מִלְכְּוָתָא וְיוֹכַח לְעַמְמִין סַגִיאִין וִיעַפְפוּן סְיָפֵיהוֹן לְסִכִּין וּמוּרְנְיָתֵיהוֹן לְמַגָלִין לָא יִטוֹל עַם בְּאַפֵּי עַם זְיַן וְלָא יִתְאַלְפוּן עוֹד מִבְּכֵן קְרָבָא׃

לֹא־יִשָּׂא גוֹי אֶל־גּוֹי חֶרֶב וְלֹא־יִלְמְדוּ עוֹד מִלְחָמָה׃
לָא יִטוֹל עַם בְּאַפֵּי עַם זְיַן וְלָא יִתְאַלְפוּן עוֹד מִבְּכֵן קְרָבָא
וְכִתְּתוּ חַרְבוֹתָם לְאִתִּים וַחֲנִיתוֹתֵיהֶם לְמַזְמֵרוֹת׃
וִיעַפְפוּן סְיָפֵיהוֹן לְסִכִּין וּמוּרְנְיָתֵיהוֹן לְמַגָלִין׃
הרגתי את־תום׃
קטלית ליה לתאמא.
אֲדַבֵּר אֶל־בִּתִּי יְהוּדִית לְבַדָּהּ׃
בַּהֲדֵי בְּרַתִּי לָא מִשְׁתָּעֵינָא אֶלָּא בִּלְשׁוֹן הַקֹּדֶשׁ.
הוּא מְבַקֵּשׁ אֶת־עֵטוֹ׃
קָא מְהַדַּר בָּתַר קוּלְמוּסָא דִּידֵיהּ.
הֲזָכָר כּוּקִי אִם־נְקֵבָה הִיא׃
כּוּקִי דְּכוּרָא הוּא אוֹ נוּקְבְתָּא הִיא?
לֹא יָסִיר הַמָּוֶת אֶת־הַצָּרָה׃
מִיתוּתָא לָא מַעְבְּרָא צַעֲרָא.
בֵּית חַשְׁמוֹנַאי הָיָה בֵּית מַלְכוּת יְהוּדִי, אֲשֶׁר מָשַׁל בְּאֶרֶץ יִשְׂרָאֵל וְעָמַד בְּרֹאשׁ מֶרֶד הַמַּכַּבִּים׃
בֵּית חַשְׁמוֹנַאי הֲוָה בֵּית מַלְכוּתָא דִּיהוּדָאֵי, דְּשַׁלִּיט בְּאַרְעָא דְּיִשְׂרָאֵל וְקָאֵי בְּרֵישׁ מַרְדּוּתָא דְּמַכַּבָּאֵי.
הֵמָּה מָגִנֵּי הָעִיר׃
אִינוּן נָטוֹרֵי מְדִינְתָּא.
